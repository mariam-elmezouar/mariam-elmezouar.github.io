Solution to Question 1:

# Initialize time to zero
time = 0.0

# Add 0.1 to time, 100,000 times
for k in range(100000):
    time += 0.1

# Check if time equals 10,000
if time == 10000.0:
    print('time to launch missile')

Explanation: The above code may not print “time to launch missile” because floating point numbers are not stored precisely in memory, leading to accumulated rounding errors. Specifically, 0.1 cannot be represented with full precision in binary. When adding it many times, the error accumulates.

Solution to Question 2:

def find_big_diag(M): 
	# Initialize the largest value and its position 
	largest_value = M[0][0] 
	position = 0 
	# Iterate over the main diagonal 
	for i in range(len(M)): 
		# Check if the current diagonal element is greater than the largest_value found so far 
		if M[i][i] > largest_value: 
			largest_value = M[i][i] 
			position = i 
	return position, largest_value

Solution to Question 3:

def best12avg(grades):
    # Initialize the sum of grades to zero
    sum_of_grades = 0
    # Initialize the minimum grade to the first grade in the list
    min_grade = grades[0]
    
    # Calculate the sum of all grades and find the minimum grade
    for grade in grades:
        sum_of_grades += grade
        if grade < min_grade:
            min_grade = grade
    
    # Subtract the minimum grade from the total sum and calculate the average
    average = (sum_of_grades - min_grade) / 12
    return average

Explanation: We iterate over the list once, summing all the grades and keeping track of the lowest grade. Then we subtract the lowest grade from the total sum and divide by 12 to find the average of the best 12 grades.

Solution to Problem 4: Matrix Transpose

def transpose_matrix(matrix):
    rows = len(matrix)
    cols = len(matrix[0])
    
    #Initialize the new matrix with transposed dimensions 
    transposed = [] 
    for i in range(cols): 
	    # Initialize a new row with zeros 
	    new_row = []
	    for j in range(rows): 
		    new_row.append(0) 
	    transposed.append(new_row)
	    
    for i in range(rows):
        for j in range(cols):
            transposed[j][i] = matrix[i][j]
    
    return transposed

# Example usage
example_matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]
print(transpose_matrix(example_matrix))

Output:

[[1, 4, 7], [2, 5, 8], [3, 6, 9]]

Solution to Problem 5: Island Perimeter

def island_perimeter(grid):
    rows = len(grid) 
    cols = len(grid[0])
    perimeter = 0

    for i in range(rows):
        for j in range(cols):
            if grid[i][j] == 1:
                if i == 0 or grid[i-1][j] == 0:
	                perimeter += 1
                if i == rows-1 or grid[i+1][j] == 0: 
	                perimeter += 1
                if j == 0 or grid[i][j-1] == 0: 
	                perimeter += 1
                if j == cols-1 or grid[i][j+1] == 0:
	                perimeter += 1

    return perimeter

# Example usage
example_grid = [
    [0, 1, 0, 0],
    [1, 1, 1, 0],
    [0, 1, 0, 0],
    [1, 1, 0, 0]
]
print(island_perimeter(example_grid))

Output:

16

Solution to Problem 6: Zero Matrix

def zero_matrix(matrix):
    rows, cols = len(matrix), len(matrix[0])
    zero_rows = set()
    zero_cols = set()

    for i in range(rows):
        for j in range(cols):
            if matrix[i][j] == 0:
                zero_rows.add(i)
                zero_cols.add(j)

    for i in zero_rows:
        for j in range(cols):
            matrix[i][j] = 0

    for j in zero_cols:
        for i in range(rows):
            matrix[i][j] = 0

    return matrix

# Example usage
example_matrix = [
    [1, 2, 3],
    [4, 0, 6],
    [7, 8, 9]
]
print(zero_matrix(example_matrix))

Output:

[[1, 0, 3], [0, 0, 0], [7, 0, 9]]
