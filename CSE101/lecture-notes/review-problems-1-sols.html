The sum_large_per_column function:


def sum_large_per_column(a_matrix):
    num_columns = len(a_matrix[0])
    new_list = []
    for i in range(num_columns):  # outer loop for columns
        col_sum = 0
        for row in a_matrix:  # inner loop over rows
            value = row[i]
            if value >= 100 or value <= -100:
                col_sum += value
        new_list.append(col_sum)
    return new_list

]

The poly_intersect function using the provided check_intersection function:


def poly_intersect(poly, lseg):
    # Loop through each vertex in the polygon
    for i in range(len(poly)):
        # Get the current vertex, pt1, and the next vertex, pt2.
        # The modulo operator (%) is used to ensure that when we reach the last vertex,
        # we loop back to the first vertex, thus closing the polygon.
        pt1, pt2 = poly[i], poly[(i+1) % len(poly)]
        
        # Check if the line segment formed by the current vertex (pt1) and the next vertex (pt2)
        # intersects with the given line segment (lseg).
        # The check_intersection function is called with the x and y coordinates of both ends
        # of the polygon's edge and the line segment.
        if check_intersection(pt1[0], pt1[1], pt2[0], pt2[1], lseg[0], lseg[1], lseg[2], lseg[3]):
            # If an intersection is found, return True immediately.
            return True
    # If no intersections are found after checking all edges, return False.
    return False
