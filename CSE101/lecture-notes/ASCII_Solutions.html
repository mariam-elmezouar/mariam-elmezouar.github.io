Solution to Caesar Cipher

def decrypt(text, shift):
    # Initialize an empty string for the result
    decrypted_text = ""
            
    # Iterate over each character in the encrypted text
    for char in text:
        # Add the shift value to each character's ASCII value to decrypt
        decrypted_text += chr((ord(char) + shift))

    return decrypted_text
            
# Decrypt the provided message 'UkqCkpEp' with a shift =  4
encrypted_message = 'UkqCkpEp'
decrypted_message = decrypt(encrypted_message, 4)
# Print the decrypted message
print("Decrypted Message:", decrypted_message)
